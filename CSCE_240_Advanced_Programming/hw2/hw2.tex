I found a really great computational geometry library that stores circles and
squares in a very particular way--by keeping up with the points representing
each object. \\
\\
I require additional functionality for a project and am tasking you with the
development of the library.  \\
\\
%
I have included a couple tests demonstrating the interfaces. You should
definitely extend these tests. I will. You will also find a comp\_geo header and
source file. I provide those as an interface specification. \textbf{YOU} are
responsible for files you edit.\\
\\
%
The functions should perform as follows:
\begin{itemize}
  \item \texttt{GetCircumference}($x_c$: double, $y_c$: double, $x_e$: double,
    $y_e$: double): double
    \begin{itemize}
      \item parameters:
        \begin{itemize}
          \item $x_{c}$: double, x-value of circle's center,
          \item $y_{c}$: double, y-value of circle's center,
          \item $x_{e}$: double, x-value of point on circle's edge, and
          \item $y_{e}$: double, y-value of point on circle's edge
        \end{itemize}

    \item \textbf{returns}: the floating point value representing the
      circumference of a circle centered at $(x_c, y_c)$ with a second point on
      the edge $(x_e, y_e)$.
  \end{itemize}

  \item \texttt{GetVolume}($x_c$: double, $y_c$: double, $x_e$: double,
    $y_e$: double): double
  \begin{itemize}
    \item parameters:
      \begin{itemize}
          \item $x_{c}$: double, x-value of circle's center,
          \item $y_{c}$: double, y-value of circle's center,
          \item $x_{e}$: double, x-value of point on circle's edge, and
          \item $y_{e}$: double, y-value of point on circle's edge
      \end{itemize}
    \item \textbf{returns}: the floating point value representing the volume of
      the circle centered at $(x_c, y_c)$ with a second point on the edge $(x_e,
      y_e)$.

  \end{itemize}

  \item \texttt{GetPerimeter}($x_{ll}$: double, $y_{ll}$: double, $x_{ur}$:
      double, $y_{ur}$: double): double
    \begin{itemize}
      \item parameters:
      \begin{itemize}
        \item $x_{ll}$: double, x-value of lower-left point of rectangle,
        \item $y_{ll}$: double, y-value of lower-left point of rectangle,
        \item $x_{ur}$: double, x-value of upper-right point of rectangle, and
        \item $y_{ur}$: double, y-value of upper-right point of rectangle
      \end{itemize}
      \item \textbf{returns}: the floating point value representing the
        perimeter of the rectangle 
    \end{itemize}
    \newpage

  \item \texttt{GetDistanceSquared}($x_1$: double, $y_1$: double, $x_2$: double,
      $y_2$: double): double
    \begin{itemize}
      \item parameters:
      \begin{itemize}
        \item $x_1$: double, x-value of point 1,
        \item $y_1$: double, y-value of point 1,
        \item $x_2$: double, x-value of point 2, and
        \item $y_2$: double, y-value of point 2
      \end{itemize}
      \item \textbf{returns}: the floating point value representing the
        squared distance between points 1 and 2.
    \end{itemize}

  \item \texttt{GetDistance}($x_1$: double, $y_1$: double, $x_2$: double,
      $y_2$: double): double
    \begin{itemize}
      \item parameters:
      \begin{itemize}
        \item $x_1$: double, x-value of point 1,
        \item $y_1$: double, y-value of point 1,
        \item $x_2$: double, x-value of point 2, and
        \item $y_2$: double, y-value of point 2
      \end{itemize}
    \item \textbf{returns}: the floating point value representing the
      distance between points 1 and 2.
  \end{itemize}
\end{itemize}
